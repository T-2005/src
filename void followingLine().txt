void followingLine() {
  P = error;
  D = error - previousError;
  PIDvalue = Kp * P  + Kd * D;
  previousError = error;

  int lsp = v - PIDvalue;
  int rsp = v + PIDvalue;

  if (lsp > 250) {
    lsp = 250;
  }
  if (lsp < 0) {
    lsp = 0;
  }
  if (rsp > 250) {
    rsp = 250;
  }
  if (rsp < 0) {
    rsp = 0;
  }
  ledcWrite(0, lsp);
  digitalWrite(AIN1, 1);
  digitalWrite(AIN2, 0);
  ledcWrite(1, rsp);  // motor_b tien
  digitalWrite(BIN1, 1);
  digitalWrite(BIN2, 0);
}
